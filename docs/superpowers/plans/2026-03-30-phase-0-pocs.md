app.post('/api/auth/register', (req, res) => {
  const { name, email, password } = req.body;
  if (!name || !email || !password) {
    return res.status(400).json({ error: 'Please fill in all fields' });
  }
  if (password.length < 8) {
    return res.status(400).json({ error: 'Password must be at least 8 characters' });
  }
  if (users[email]) {
    return res.status(409).json({ error: 'Email already registered' });
  }
  const userId = crypto.randomUUID();
  users[email] = { id: userId, name, email };
  const sessionId = crypto.randomUUID();
  sessions[sessionId] = userId;
  res.cookie('session', sessionId, { httpOnly: true, path: '/' });
  res.status(201).json({ user_id: userId });
});

app.get('/api/user/me', (req, res) => {
  const sessionId = req.cookies?.session;
  if (!sessionId || !sessions[sessionId]) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  const userId = sessions[sessionId];
  const user = Object.values(users).find(u => u.id === userId);
  if (!user) return res.status(404).json({ error: 'User not found' });
  res.json({ name: user.name, email: user.email });
});

## Task 3: PoC-1 — Inspect AgentHistoryList structure

**Files:**
- Create: `poc/poc1_navigator_output.py`

This script runs browser-use against the test app, then inspects every field of the returned `AgentHistoryList` to confirm what structured data is available for the v3 output schema.

- [ ] **Step 1: Write the PoC script**

```python
"""PoC-1: Inspect browser-use AgentHistoryList structure.

Runs browser-use against the test app, then inspects the returned
AgentHistoryList to map available fields to the v3 output schema.

Prerequisites:
  - pip install browser-use langchain-openai
  - OPENROUTER_API_KEY env var set
  - Test app running: cd poc/test_app && node server.js

Usage:
  python poc/poc1_navigator_output.py
"""

import asyncio
import json
import os
import sys
from pathlib import Path

# Add parent dir so we can import persona_browser config
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))


async def main():
    from browser_use import Agent, BrowserSession, BrowserProfile
    from langchain_openai import ChatOpenAI

    # --- Config ---
    URL = "http://localhost:3333/register"
    TASK = (
        "You are Jordan Rivera, a first-time visitor. "
        "Go to the registration page. Fill the signup form with: "
        "name='Jordan Rivera', email='jordan@example.com', password='SecurePass1'. "
        "Submit the form. After redirect, report what you see on the dashboard."
    )

    api_key = os.environ.get("OPENROUTER_API_KEY", "")
    if not api_key:
        print("ERROR: Set OPENROUTER_API_KEY environment variable")
        sys.exit(1)

    llm = ChatOpenAI(
        model="google/gemini-2.5-flash-preview",
        api_key=api_key,
        base_url="https://openrouter.ai/api/v1",
        temperature=0.1,
    )

    profile = BrowserProfile(
        headless=False,  # visible for debugging during PoC
        viewport={"width": 1280, "height": 720},
    )
    session = BrowserSession(browser_profile=profile)

    print("=" * 70)
    print("PoC-1: AgentHistoryList Inspection")
    print("=" * 70)
    print(f"URL: {URL}")
    print(f"Task: {TASK[:80]}...")
    print()

    agent = Agent(task=TASK, llm=llm, browser_session=session, max_steps=20)

    try:
        result = await agent.run()
    finally:
        await session.stop()

    # --- Inspect the result ---
    print("\n" + "=" * 70)
    print("RESULT TYPE:", type(result).__name__)
    print("=" * 70)

    # 1. Basic status
    print(f"\n--- Basic Status ---")
    print(f"  is_done(): {result.is_done()}")
    if hasattr(result, "is_successful"):
        print(f"  is_successful(): {result.is_successful()}")
    print(f"  total_duration_seconds(): {result.total_duration_seconds():.1f}s")
    print(f"  number_of_steps(): {result.number_of_steps()}")

    # 2. URLs per step
    print(f"\n--- URLs per step ---")
    urls = result.urls()
    for i, url in enumerate(urls):
        print(f"  Step {i}: {url}")

    # 3. Actions per step
    print(f"\n--- Actions per step ---")
    action_names = result.action_names()
    for i, name in enumerate(action_names):
        print(f"  Step {i}: {name}")

    # 4. Full action details
    print(f"\n--- Model Actions (full params) ---")
    model_actions = result.model_actions()
    for i, action in enumerate(model_actions):
        print(f"  Step {i}: {json.dumps(action, default=str)[:200]}")

    # 5. Action results
    print(f"\n--- Action Results ---")
    action_results = result.action_results()
    for i, ar in enumerate(action_results):
        print(f"  Result {i}: {str(ar)[:200]}")

    # 6. Screenshots
    print(f"\n--- Screenshot Paths ---")
    screenshot_paths = result.screenshot_paths()
    for i, path in enumerate(screenshot_paths):
        exists = Path(path).exists() if path else False
        print(f"  Step {i}: {path} (exists={exists})")

    # 7. Model thoughts (LLM reasoning)
    print(f"\n--- Model Thoughts (per step) ---")
    thoughts = result.model_thoughts()
    for i, thought in enumerate(thoughts):
        thought_str = str(thought)[:300]
        print(f"  Step {i}: {thought_str}")

    # 8. Extracted content
    print(f"\n--- Extracted Content ---")
    content = result.extracted_content()
    for i, c in enumerate(content):
        if c:
            print(f"  Step {i}: {c[:200]}")

    # 9. Errors
    print(f"\n--- Errors ---")
    errors = result.errors()
    for i, err in enumerate(errors):
        if err:
            print(f"  Step {i}: {err}")
    if not any(errors):
        print("  (none)")

    # 10. Final result
    print(f"\n--- Final Result ---")
    final = result.final_result()
    if final:
        print(f"  {final[:500]}")
    else:
        print("  (none)")

    # 11. Per-step history inspection
    print(f"\n--- Per-Step History Detail ---")
    for i, step in enumerate(result.history):
        print(f"\n  Step {i}:")
        print(f"    type: {type(step).__name__}")
        # BrowserStateHistory
        if hasattr(step, "state") and step.state:
            state = step.state
            print(f"    state.url: {getattr(state, 'url', 'N/A')}")
            print(f"    state.title: {getattr(state, 'title', 'N/A')}")
            if hasattr(state, "tabs"):
                print(f"    state.tabs: {state.tabs}")
            if hasattr(state, "screenshot_path"):
                print(f"    state.screenshot_path: {state.screenshot_path}")
            if hasattr(state, "interacted_element"):
                elem = state.interacted_element
                if elem:
                    print(f"    state.interacted_element: {str(elem)[:200]}")

    # 12. Save to file
    output_path = Path(__file__).parent / "poc1_output.json"
    result.save_to_file(str(output_path))
    print(f"\n--- Saved full history to {output_path} ---")
    file_size = output_path.stat().st_size
    print(f"  File size: {file_size / 1024:.1f} KB")

    # 13. V3 Schema mapping assessment
    print("\n" + "=" * 70)
    print("V3 SCHEMA MAPPING ASSESSMENT")
    print("=" * 70)

    unique_urls = list(dict.fromkeys(u for u in urls if u))
    print(f"\n  Unique URLs visited: {unique_urls}")
    print(f"  → pages[] can be derived by grouping steps by URL: {'YES' if len(unique_urls) > 1 else 'NEEDS TESTING'}")
    print(f"  → pages[].url_visited: {'YES' if urls else 'NO'} (from urls())")
    print(f"  → pages[].screenshot: {'YES' if any(screenshot_paths) else 'NO'} (from screenshot_paths())")
    print(f"  → pages[].observations.actions[]: {'YES' if action_names else 'NO'} (from action_names() + model_actions())")
    print(f"  → pages[].observations.description: {'PARTIAL' if thoughts else 'NO'} (from model_thoughts() — LLM text)")
    print(f"  → experience: {'YES' if final else 'PARTIAL'} (from final_result())")
    print(f"  → elapsed_seconds: YES (from total_duration_seconds())")
    print(f"  → network_log: SEPARATE (from HAR file — see PoC-2)")

    print(f"\n  VERDICT: ", end="")
    if urls and action_names and screenshot_paths:
        print("OUTCOME A — deterministic parser feasible")
    elif urls and action_names:
        print("OUTCOME A/B — parser feasible, screenshots may need config")
    else:
        print("OUTCOME B/C — needs further investigation")

    print("\nPoC-1 complete.")


if __name__ == "__main__":
    asyncio.run(main())
```

## Task 4: PoC-2 — Validate HAR network capture

**Files:**
- Create: `poc/poc2_network_capture.py`

This script runs browser-use with HAR recording enabled, then parses the HAR file to verify it captures the network data needed for the v3 pipeline.

- [ ] **Step 1: Write the PoC script**

```python
"""PoC-2: Validate HAR network capture during browser-use session.

Runs browser-use with BrowserProfile.record_har_path enabled,
then parses the HAR file to verify it captures the data needed
for the v3 network_log[] schema.

Prerequisites:
  - pip install browser-use langchain-openai
  - OPENROUTER_API_KEY env var set
  - Test app running: cd poc/test_app && node server.js

Usage:
  python poc/poc2_network_capture.py
"""

import asyncio
import json
import os
import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

HAR_PATH = Path(__file__).parent / "session.har"


async def main():
    from browser_use import Agent, BrowserSession, BrowserProfile
    from langchain_openai import ChatOpenAI

    # --- Config ---
    URL = "http://localhost:3333/register"
    TASK = (
        "You are Jordan Rivera, a first-time visitor. "
        "Go to the registration page. Fill the signup form with: "
        "name='Jordan Rivera', email='jordan@example.com', password='SecurePass1'. "
        "Submit the form. After redirect, report what you see on the dashboard. "
        "Then refresh the page and confirm the data is still there."
    )

    api_key = os.environ.get("OPENROUTER_API_KEY", "")
    if not api_key:
        print("ERROR: Set OPENROUTER_API_KEY environment variable")
        sys.exit(1)

    llm = ChatOpenAI(
        model="google/gemini-2.5-flash-preview",
        api_key=api_key,
        base_url="https://openrouter.ai/api/v1",
        temperature=0.1,
    )

    # Clean up previous HAR
    if HAR_PATH.exists():
        HAR_PATH.unlink()

    profile = BrowserProfile(
        headless=False,
        viewport={"width": 1280, "height": 720},
        record_har_path=str(HAR_PATH),
        record_har_content="embed",
    )
    session = BrowserSession(browser_profile=profile)

    print("=" * 70)
    print("PoC-2: HAR Network Capture Validation")
    print("=" * 70)
    print(f"URL: {URL}")
    print(f"HAR output: {HAR_PATH}")
    print()

    agent = Agent(task=TASK, llm=llm, browser_session=session, max_steps=20)

    try:
        result = await agent.run()
        print(f"\nAgent completed in {result.total_duration_seconds():.1f}s")
        print(f"Steps: {result.number_of_steps()}")
    finally:
        await session.stop()

    # --- Parse HAR file ---
    print("\n" + "=" * 70)
    print("HAR FILE ANALYSIS")
    print("=" * 70)

    if not HAR_PATH.exists():
        print("ERROR: HAR file was not created!")
        print("This means BrowserProfile.record_har_path did not work.")
        print("Fallback needed: Playwright-via-CDP or custom CDP watchdog.")
        sys.exit(1)

    har_size = HAR_PATH.stat().st_size
    print(f"\nHAR file size: {har_size / 1024:.1f} KB")

    with open(HAR_PATH, "r", encoding="utf-8") as f:
        har = json.load(f)

    entries = har.get("log", {}).get("entries", [])
    print(f"Total entries: {len(entries)}")

    # --- Analyze each entry ---
    print(f"\n--- All HTTP Entries ---")
    for i, entry in enumerate(entries):
        req = entry.get("request", {})
        resp = entry.get("response", {})
        method = req.get("method", "?")
        url = req.get("url", "?")
        status = resp.get("status", "?")
        timing = entry.get("time", 0)

        # Shorten URL for display
        display_url = url
        if "localhost:3333" in url:
            display_url = url.split("localhost:3333")[-1] or "/"

        print(f"  [{i}] {method} {display_url} -> {status} ({timing:.0f}ms)")

    # --- Check for API calls ---
    print(f"\n--- API Calls (matching /api/) ---")
    api_entries = [e for e in entries if "/api/" in e.get("request", {}).get("url", "")]
    for i, entry in enumerate(api_entries):
        req = entry.get("request", {})
        resp = entry.get("response", {})
        method = req.get("method")
        url = req.get("url")
        status = resp.get("status")
        path = url.split("localhost:3333")[-1] if "localhost:3333" in url else url

        print(f"\n  API Call {i}: {method} {path} -> {status}")

        # Request headers
        req_headers = {h["name"].lower(): h["value"] for h in req.get("headers", [])}
        print(f"    Content-Type: {req_headers.get('content-type', 'N/A')}")
        if "cookie" in req_headers:
            print(f"    Cookie: {req_headers['cookie'][:80]}")
        else:
            print(f"    Cookie: (none)")

        # Request body (POST data)
        post_data = req.get("postData", {})
        if post_data:
            mime = post_data.get("mimeType", "")
            text = post_data.get("text", "")
            print(f"    Request body ({mime}): {text[:200]}")

        # Response headers
        resp_headers = {h["name"].lower(): h["value"] for h in resp.get("headers", [])}
        if "set-cookie" in resp_headers:
            print(f"    Set-Cookie: {resp_headers['set-cookie'][:100]}")

        # Response body
        resp_content = resp.get("content", {})
        resp_text = resp_content.get("text", "")
        resp_mime = resp_content.get("mimeType", "")
        if resp_text:
            print(f"    Response body ({resp_mime}): {resp_text[:200]}")

    # --- V3 network_log mapping ---
    print("\n" + "=" * 70)
    print("V3 NETWORK_LOG MAPPING ASSESSMENT")
    print("=" * 70)

    checks = {
        "HAR file created": HAR_PATH.exists(),
        "Contains entries": len(entries) > 0,
        "API calls captured": len(api_entries) > 0,
        "Method captured": all(e.get("request", {}).get("method") for e in api_entries),
        "URL captured": all(e.get("request", {}).get("url") for e in api_entries),
        "Status captured": all(e.get("response", {}).get("status") for e in api_entries),
        "Timing captured": all(e.get("time", 0) > 0 for e in api_entries),
    }

    # Check for auth cookie capture
    post_register = [e for e in api_entries
                     if "register" in e.get("request", {}).get("url", "")
                     and e.get("request", {}).get("method") == "POST"]
    if post_register:
        resp_headers = {h["name"].lower(): h["value"]
                       for h in post_register[0].get("response", {}).get("headers", [])}
        checks["Set-Cookie captured on register"] = "set-cookie" in resp_headers

    get_me = [e for e in api_entries
              if "user/me" in e.get("request", {}).get("url", "")]
    if get_me:
        req_headers = {h["name"].lower(): h["value"]
                      for h in get_me[0].get("request", {}).get("headers", [])}
        checks["Cookie sent on /user/me"] = "cookie" in req_headers

    # Request body
    if post_register:
        post_data = post_register[0].get("request", {}).get("postData", {})
        checks["Request body captured (POST)"] = bool(post_data.get("text"))

    # Response body
    if get_me:
        resp_content = get_me[0].get("response", {}).get("content", {})
        checks["Response body captured"] = bool(resp_content.get("text"))

    print()
    all_pass = True
    for check, passed in checks.items():
        status = "PASS" if passed else "FAIL"
        if not passed:
            all_pass = False
        print(f"  [{status}] {check}")

    # --- Edge cases ---
    print(f"\n--- Edge Case Analysis ---")

    # CORS preflight
    options_entries = [e for e in entries if e.get("request", {}).get("method") == "OPTIONS"]
    print(f"  OPTIONS (preflight) requests: {len(options_entries)}")

    # Redirects
    redirect_entries = [e for e in entries if 300 <= e.get("response", {}).get("status", 0) < 400]
    print(f"  Redirect responses (3xx): {len(redirect_entries)}")
    for r in redirect_entries:
        url = r.get("request", {}).get("url", "")
        status = r.get("response", {}).get("status")
        print(f"    {status} {url}")

    # Concurrent requests (same startedDateTime)
    timestamps = [e.get("startedDateTime", "") for e in entries]
    from collections import Counter
    ts_counts = Counter(timestamps)
    concurrent = {ts: count for ts, count in ts_counts.items() if count > 1}
    print(f"  Concurrent requests (same timestamp): {len(concurrent)} groups")

    # File size concern
    if har_size > 1_000_000:
        print(f"  WARNING: HAR file is {har_size / 1024 / 1024:.1f} MB — consider record_har_content='omit'")
    else:
        print(f"  HAR file size OK ({har_size / 1024:.1f} KB)")

    # --- Verdict ---
    print(f"\n  VERDICT: ", end="")
    if all_pass:
        print("HAR recording fully functional — proceed with har_parser.py")
    else:
        failed = [k for k, v in checks.items() if not v]
        print(f"PARTIAL — failed: {failed}")
        print("  Consider Playwright-via-CDP fallback for missing data")

    print("\nPoC-2 complete.")


if __name__ == "__main__":
    asyncio.run(main())
```
